def sinus(w, phi, A, dy):
    global sin
    sin = 0
    xy = []
    for x in range(1000):
        y=math.sin(x * w)
        xy.append(x + phi)
        xy.append(y * A + dy)
    sin = canvas.create_line(xy, fill='blue')

def clean():
    canvas.delete(sin)

btn_calc = Button(root, text='Порахувати')
